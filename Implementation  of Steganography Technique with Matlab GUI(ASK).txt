function sonuc=fonk(A)


 degisken1=1; 
for i=1:length(A)
   degisken=1;
  
   binofA=dectobin(A(i));
  
    
    for degisken=1:8
        
      Abin(degisken1)=binofA(degisken);
        degisken1=degisken1+1;

function sonuc=dectobin(D)
B = zeros(1,8);
for k = 8:-1:1
  B(k) = mod(D,2);
  D = (D-B(k))/2;
end

sonuc=B;





function sonuc=bintodec(D)

sonuc= sum(D.*(2.^(7:-1:0)));

kontrol=0;

 j=j+1;

function sonuc=cikarma(A,B,N)


j=1;
kontrol=1;
sizeofB=size(B);

for i=1:length(A)

    if j<=sizeofB(1)*sizeofB(2)
        if kontrol==N
        kontrol=0;
        sonuc(j)=A(i);
        j=j+1;
    
        end

    end
kontrol=kontrol+1;
end







function sonuc=matrixyap(A,i,j)

say=1;
B=zeros(i,j);

for n=1:i
    for m=1:j
      B(n,m)=A(say);
      say=say+1;
    end
    
    
end

sonuc=B;
